from turtle import Turtle, Screen

wn = Screen()

def tree(length,t):
    if length > 5:
        t.forward(length) # ileri git
        t.right(40)       # saða dön
        tree(length-15,t) # daha küçük bir aðaç çiz
        t.left(80)        # sola dön
        tree(length-15,t) # daha küçük bir aðaç çiz
        t.right(40)       # Baþladýðýn açýya geri dön
        t.backward(length)# Baþladýðýn noktaya geri dön


t = Turtle()
t.left(90) # Yukarýya doðru çizmek için
t.speed(0) # en hýzlý animasyon

tree(100,t)

wn.exitonclick()
